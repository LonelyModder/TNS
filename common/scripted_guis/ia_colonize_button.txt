ia_colonize_button = {
	scope = province
	confirm_title = "ia_colonize_button_confirm_t"    
    confirm_text = "ia_colonize_button_confirm_desc" 
	saved_scopes = {
		player
	}
	ai_is_valid = {
		scope:player = {
			manpower >= 2
		}
	}

	is_valid = {
		scope:player = {
			manpower >= 1
		}
	}
	effect = {
		scope:player = {
			pay_price = price_colonize
			add_manpower = -1
		}
		if = {
			limit = {
				has_owner = no
			}
			set_owned_by = scope:player
		}
		colonize_effect = yes
		#pay_colonize_effect = yes
		every_neighbor_province = {
			limit = {
				has_owner = yes
				NOT = {
					owner = scope:player
				}
			}
			root = {
				add_claim = prev.owner
			}
		}
	}
}
